        function initSidebar() {
            const sidebar = document.getElementById('sidebar');
            const overlay = document.getElementById('sidebar-overlay');
            const btnOpen = document.getElementById('btn-toggle-sidebar');
            const btnClose = document.getElementById('btn-close-sidebar');

            const open = () => {
                sidebar.classList.remove('-translate-x-full');
                overlay.classList.remove('hidden');
                setTimeout(() => overlay.classList.add('opacity-100'), 10);
            };
            const close = () => {
                sidebar.classList.add('-translate-x-full');
                overlay.classList.remove('opacity-100');
                setTimeout(() => overlay.classList.add('hidden'), 300);
            };

            btnOpen.onclick = open;
            btnClose.onclick = close;
            overlay.onclick = close;
        }

            const container = document.getElementById('color-inputs-container');
            container.innerHTML = brandState.colors.map((c, i) => `
                <div class="flex items-center gap-3">
                    <div class="w-10 h-10 rounded-xl overflow-hidden border border-gray-100 shrink-0">
                        <input type="color" value="${c}" data-idx="${i}" class="c-pick">
                    </div>
                    <input type="text" value="${c}" data-idx="${i}" class="c-text w-full p-2 bg-gray-50 rounded-xl text-xs font-mono uppercase outline-none">
                </div>
            `).join('');

            document.querySelectorAll('.c-pick, .c-text').forEach(el => {
                el.oninput = (e) => {
                    brandState.colors[e.target.dataset.idx] = e.target.value;
                    renderColorInputs(); updateUI();
                };
            });
        }

        function renderMoodboardInputs() {
            const container = document.getElementById('moodboard-inputs-container');
            container.innerHTML = brandState.moodboard.map((img, i) => `
                <div class="relative">
                    <input type="file" id="upload-mood-input-${i}" class="hidden mood-in" accept="image/*" data-idx="${i}">
                    <div onclick="document.getElementById('upload-mood-input-${i}').click()" class="w-full h-24 border-2 border-dashed border-gray-100 rounded-2xl flex items-center justify-center cursor-pointer hover:bg-white transition-all overflow-hidden bg-gray-50">
                        ${img ? `<img src="${img}" class="w-full h-full object-cover">` : `<i data-lucide="image-plus" class="w-6 h-6 text-gray-300"></i>`}
                    </div>
                </div>
            `).join('');
            lucide.createIcons();
            document.querySelectorAll('.mood-in').forEach(el => {
                el.onchange = (e) => {
                    const f = e.target.files[0];
                    if(f) {
                        brandState.moodboard[e.target.dataset.idx] = URL.createObjectURL(f);
                        renderMoodboardInputs(); updateUI();
                    }
                };
            });
        }

        function updateUI() {
            document.getElementById('input-name').value = brandState.name;
            document.getElementById('input-desc').value = brandState.description;
            document.getElementById('preview-name').innerText = brandState.name || 'Sin Nombre';
            document.getElementById('preview-desc').innerText = `"${brandState.description || 'Sin descripción'}"`;
            document.getElementById('preview-style-label').innerText = brandState.style;

            ['primary', 'secondary'].forEach(t => {
                const f = brandState.fonts[t];
                if(!f.startsWith('CustomFont')) {
                    const l = document.createElement('link');
                    l.href = `https://fonts.googleapis.com/css2?family=${f.replace(/ /g, '+')}:wght@400;700&display=swap`;
                    l.rel = 'stylesheet'; document.head.appendChild(l);
                }
                document.querySelectorAll(`.font-${t}`).forEach(el => el.style.fontFamily = `"${f}", sans-serif`);
                document.getElementById(`preview-font-${t}-name`).innerText = f.startsWith('CustomFont') ? 'Fuente Especial' : f;
            });

            document.getElementById('preview-colors-container').innerHTML = brandState.colors.map(c => `
                <div class="flex-1 flex flex-col gap-2">
                    <div class="w-full aspect-square rounded-2xl shadow-inner border border-black/5" style="background-color: ${c}"></div>
                    <span class="text-center font-mono text-[9px] text-gray-400 uppercase">${c}</span>
                </div>
            `).join('');

            const updateImg = (id, src, h, iconSize) => {
                const el = document.getElementById(id);
                if(src) el.innerHTML = `<img src="${src}" class="${h} object-contain rounded-xl mx-auto">`;
                else el.innerHTML = `<i data-lucide="image-plus" class="${iconSize} text-gray-200"></i>`;
            };

            // Lógica especial para el logo del encabezado (Header)
            const headerLogoEl = document.getElementById('preview-header-logo-container');
            const headerSrc = brandState.logos.primary || DEMO_LOGO;
            headerLogoEl.innerHTML = `<img src="${headerSrc}" class="h-20 object-contain mx-auto">`;

            // Lógica normal para el resto de logos en la cuadrícula
            updateImg('preview-logo-primary-container', brandState.logos.primary, 'max-h-40', 'w-10 h-10');
            updateImg('preview-logo-secondary-container', brandState.logos.secondary, 'max-h-32', 'w-10 h-10');
            updateImg('preview-logo-tertiary-container', brandState.logos.tertiary, 'max-h-32', 'w-10 h-10');
            updateImg('preview-logo-quaternary-container', brandState.logos.quaternary, 'max-h-32', 'w-10 h-10');

            brandState.moodboard.forEach((m, i) => updateImg(`preview-mood-${i}-container`, m, 'w-full h-full object-cover', 'w-10 h-10'));

            let p = 0;
            if(brandState.name) p+=10; if(brandState.description) p+=10;
            if(brandState.logos.primary) p+=25; if(brandState.logos.secondary) p+=15;
            if(brandState.colors.some(c => c !== "#ffffff")) p+=20;
            if(brandState.fonts.primary) p+=20;
            brandState.progress = Math.min(100, p);
            document.getElementById('progress-text').innerText = brandState.progress;
            document.getElementById('progress-bar').style.width = brandState.progress + '%';
            
            lucide.createIcons();
        }

        function setupEventListeners() {
            document.getElementById('input-name').oninput = (e) => { brandState.name = e.target.value; updateUI(); };
            document.getElementById('input-desc').oninput = (e) => { brandState.description = e.target.value; updateUI(); };
            document.getElementById('input-style').onchange = (e) => {
                const s = e.target.value;
                brandState.style = s;
                brandState.colors = [...stylePresets[s].colors];
                brandState.fonts = {...stylePresets[s].fonts};
                renderColorInputs(); updateUI();
            };

            ['primary', 'secondary'].forEach(t => {
                document.getElementById(`select-font-${t}`).onchange = (e) => {
                    brandState.fonts[t] = e.target.value; updateUI();
                };
                document.getElementById(`upload-font-${t}`).onchange = (e) => {
                    const f = e.target.files[0];
                    if(f) {
                        const url = URL.createObjectURL(f);
                        const name = `CustomFont_${t}_${Date.now()}`;
                        const style = document.createElement('style');
                        style.innerHTML = `@font-face { font-family: '${name}'; src: url('${url}'); }`;
                        document.head.appendChild(style);
                        brandState.fonts[t] = name; updateUI();
                    }
                };
            });

            ['primary', 'secondary', 'tertiary', 'quaternary'].forEach(t => {
                document.getElementById(`upload-logo-${t}`).onchange = (e) => {
                    const f = e.target.files[0];
                    if(f) { brandState.logos[t] = URL.createObjectURL(f); updateUI(); }
                };
            });

            document.getElementById('btn-export-pdf').onclick = () => {
                const opt = { margin:0, filename: 'BrandingBook.pdf', image:{type:'jpeg',quality:1}, html2canvas:{scale:2, useCORS:true}, jsPDF:{unit:'in', format:'letter', orientation:'portrait'} };
                html2pdf().set(opt).from(document.getElementById('preview-canvas')).save();
            };

            document.getElementById('btn-export-png').onclick = () => {
                html2canvas(document.getElementById('preview-canvas'), {scale:2, useCORS:true}).then(c => {
                    const a = document.createElement('a'); a.download = 'BrandBoard.png'; a.href = c.toDataURL('image/png'); a.click();
                });
            };

            document.getElementById('btn-save').onclick = async () => {
                const btnText = document.getElementById('btn-save-text');
                const originalBtnText = btnText.innerText;
                btnText.innerText = "Guardando...";
                try {
                    if (VAULT_WEB_APP_URL === "URL_DE_TU_WEB_APP_SCRIPT_AQUI") {
                        alert("Por favor, configura la URL de tu Vault Web App.");
                        btnText.innerText = originalBtnText;
                        return;
                    }

                    const payload = { 
                        project: "tiendas-digitales", 
                        action: "saveBranding", 
                        params: { 
                            user_id: localStorage.getItem('userId') || "USER_DEMO", 
                            progress: brandState.progress + "%", 
                            brand_data_json: JSON.stringify(brandState), 
                            last_update: new Date().toISOString() 
                        } 
                    };
                    const res = await fetch(VAULT_WEB_APP_URL, { 
                        method: 'POST', 
                        body: JSON.stringify(payload), 
                        headers: { 'Content-Type': 'text/plain;charset=utf-8' } 
                    });
                    const data = await res.json();
                    if(data.success) alert("¡Branding guardado exitosamente!");
                    else alert("Error al guardar en el servidor.");
                } catch(e) { 
                    alert("Error de conexión con el Vault."); 
                } finally { 
                    btnText.innerText = originalBtnText; 
                }
            };
        }
    </script>
