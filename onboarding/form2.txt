        // Toggle VOIP Pitch
        function toggleVoipPitch(val) {
            const pitch = document.getElementById('voip-pitch-container');
            if (val === 'No') {
                pitch.classList.remove('hidden');
            } else {
                pitch.classList.add('hidden');
            }
        }

        // Radio Styling Logic
        document.querySelectorAll('.option-card').forEach(card => {
            const radio = card.querySelector('input[type="radio"]');
            card.addEventListener('click', () => {
                const name = radio.getAttribute('name');
                document.querySelectorAll(`input[name="${name}"]`).forEach(r => {
                    r.parentElement.classList.remove('selected');
                });
                radio.checked = true;
                card.classList.add('selected');
            });
        });

        logoInput.addEventListener('change', (e) => selectedFile = e.target.files[0]);

        nextBtn.addEventListener('click', () => {
            if (validateStep(currentStep)) {
                if (currentStep < totalSteps) {
                    currentStep++;
                    updateUI();
                } else {
                    submitForm();
                }
            }
        });

        prevBtn.addEventListener('click', () => {
            if (currentStep > 1) {
                currentStep--;
                updateUI();
            }
        });

        function updateUI() {
            steps.forEach(step => step.classList.toggle('active', parseInt(step.dataset.step) === currentStep));
            dots.forEach(dot => dot.classList.toggle('active', parseInt(dot.dataset.step) <= currentStep));
            prevBtn.style.visibility = currentStep === 1 ? 'hidden' : 'visible';
            nextBtn.innerText = currentStep === totalSteps ? 'Finalizar' : 'Siguiente';
        }

        function validateStep(stepNum) {
            const currentStepEl = document.querySelector(`.step-content[data-step="${stepNum}"]`);
            const inputs = currentStepEl.querySelectorAll('input[required], select[required]');
            let isValid = true;

            inputs.forEach(input => {
                if (input.type === 'radio') {
                    const name = input.getAttribute('name');
                    if (!currentStepEl.querySelector(`input[name="${name}"]:checked`)) isValid = false;
                } else if (!input.value.trim()) {
                    isValid = false;
                    input.style.borderColor = '#ef4444';
                } else {
                    input.style.borderColor = '#e2e8f0';
                }
            });
            return isValid;
        }

        async function uploadToImageKit(file) {
            return new Promise((resolve) => {
                const reader = new FileReader();
                reader.onloadend = () => resolve(reader.result);
                reader.readAsDataURL(file);
            });
        }

            const exec = parseInt(formData.get('execution') || 1);
            
            const minScore = Math.min(inv, exec);
            let level = minScore === 3 ? "Advanced" : minScore === 2 ? "Intermediate" : "Beginner";

            return {
                level,
                scores: {
                    investment: inv === 3 ? 'Advanced' : inv === 2 ? 'Intermediate' : 'Beginner',
                    execution: exec === 3 ? 'Advanced' : exec === 2 ? 'Intermediate' : 'Beginner'
                }
            };
        }

        async function submitForm() {
            loadingOverlay.classList.remove('hidden');
            const classification = calculateClassification();

            const formData = new FormData(form);
            const payload = Object.fromEntries(formData);
            
            // Collect multi-select values
            payload.payments = Array.from(formData.getAll('payments')).join(', ');
            payload.delivery = Array.from(formData.getAll('delivery')).join(', ');
            
            payload.classification_level = classification.level;
            payload.score_details = JSON.stringify(classification.scores);
            payload.timestamp = new Date().toISOString();
            payload.logo_url = ""; 

            try {
                // 1. Upload Logo via Secure Vault (if selected)
                if (selectedFile) {
                    document.getElementById('loading-text').innerText = "Subiendo imagen segura...";
                    const base64File = await uploadToImageKit(selectedFile);

                    // Sending action 'uploadImage' to Master Vault
                    const uploadReq = await fetch(VAULT_URL, { 
                        method: 'POST', 
                        headers: { 'Content-Type': 'text/plain;charset=utf-8' }, // Ensures Google Apps Script reads JSON without CORS preflight block
                        body: JSON.stringify({ 
                            action: 'uploadImage', 
                            project: PROJECT_NAME,
                            params: {
                                file: base64File,
                                fileName: selectedFile.name,
                                folder: 'logos'
                            }
                        }) 
                    });

                    const uploadRes = await uploadReq.json();
                    if (uploadRes.success && uploadRes.data && uploadRes.data.url) {
                        payload.logo_url = uploadRes.data.url;
                    }
                }

                // 2. Sync Form Data via Secure Vault
                document.getElementById('loading-text').innerText = "Guardando perfil operativo...";
                await fetch(VAULT_URL, { 
                    method: 'POST', 
                    headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                    body: JSON.stringify({
                        action: 'submitForm',
                        project: PROJECT_NAME,
                        params: payload
                    }) 
                });

                // 3. Send Notifications via Master Vault (EmailJS hidden inside)
                document.getElementById('loading-text').innerText = "Enviando confirmaciones...";
                const templateParams = { ...payload, to_name: payload.name, to_email: payload.email };
                
                // Send User Email
                await fetch(VAULT_URL, {
                    method: 'POST',
                    headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                    body: JSON.stringify({
                        action: 'sendEmail',
                        project: PROJECT_NAME,
                        params: {
                            provider: 'emailjs',
                            to: payload.email,
                            template: 'tienda-digital-success', // Your user template ID
                            data: templateParams
                        }
                    })
                });

                // Send Admin Notification Email
                await fetch(VAULT_URL, {
                    method: 'POST',
                    headers: { 'Content-Type': 'text/plain;charset=utf-8' },
                    body: JSON.stringify({
                        action: 'sendEmail',
                        project: PROJECT_NAME,
                        params: {
                            provider: 'emailjs',
                            to: 'admin@bebell.digital', // Adjust admin receiving email if needed
                            template: 'tienda-digital-new-user', // Your admin template ID
                            data: templateParams
                        }
                    })
                });
                
            } catch (e) { 
                console.error("Vault interaction error:", e); 
            }

            loadingOverlay.classList.add('hidden');
            mainContainer.style.display = 'none';
            resultScreen.style.display = 'block';

            // Confetti animation
            const duration = 5 * 1000;
            const end = Date.now() + duration;
            (function frame() {
              confetti({ particleCount: 3, angle: 60, spread: 55, origin: { x: 0 }, colors: ['#8b84f5', '#e07dd1'] });
              confetti({ particleCount: 3, angle: 120, spread: 55, origin: { x: 1 }, colors: ['#8b84f5', '#e07dd1'] });
              if (Date.now() < end) requestAnimationFrame(frame);
            }());

            const badge = document.getElementById('result-badge');
            badge.innerText = classification.level;
            badge.className = `level-badge badge-${classification.level.toLowerCase()}`;
            document.getElementById('score-inv').innerText = classification.scores.investment;
            document.getElementById('score-exec').innerText = classification.scores.execution;

            // Auto-redirect after 5 seconds
            setTimeout(() => { window.location.href = REDIRECT_URL; }, 5000);
        }
